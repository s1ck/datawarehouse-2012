-- 1) Liste der fuenf am haeufigsten zitierten Publikationen fuer jede der
--    betrachteten Venue-Serien.

-- ACM (ACM Trans. Database Syst.)
WITH
	MEMBER [Measures].[Citings All] AS 
	'[Measures].[Citings ACM] + [Measures].[Citings GS]'
SELECT
	[Measures].[Citings All] ON 0,
	Topcount(
		[Title].[Title].Children,
		5,
		[Measures].[Citings All]
	) ON 1
FROM
	Dwhprak00
WHERE
	[VenueSeries].[Description].[ACM Trans. Database Syst.]
